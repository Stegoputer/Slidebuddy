                {"heading": "str", "text": "str"},
                {"heading": "str", "text": "str"}
            ]
        },
        {
            "heading": "str",
            "bullets": [
                {"heading": "str", "text": "str"},
                {"heading": "str", "text": "str"}
            ]
        }
    ]
}

HINWEISE:
- Exakt 2 Sektionen
- Pro Sektion 2-4 Bullet-Points
- Bullet-Headings: 2-5 Wörter, Bullet-Text: 1-2 Sätze
